  <script>
    function revealSurprise() {
      // 1. Hide the gift box
      document.getElementById('giftSection').classList.add('hide');

      // 2. Reveal the secret message card
      const card = document.getElementById('surpriseCard');
      card.classList.add('reveal');

      // 3. Play music if you added one
      const music = document.getElementById('bgMusic');
      if(music) { music.play(); }

      // 4. Trigger the Confetti Explosion!
      var duration = 4 * 1000;
      var end = Date.now() + duration;

      (function frame() {
        // Launch confetti from the left edge
        confetti({
          particleCount: 3,
          angle: 60,
          spread: 55,
          origin: { x: 0, y: 0.8 }
        });
        // Launch confetti from the right edge
        confetti({
          particleCount: 3,
          angle: 120,
          spread: 55,
          origin: { x: 1, y: 0.8 }
        });

        if (Date.now() < end) {
          requestAnimationFrame(frame);
        }
      }());
    }
  </script>
